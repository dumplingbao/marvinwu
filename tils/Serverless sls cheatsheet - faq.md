-  how to give s3 access to lambda?

	```
	  name: aws
	  runtime: python3.6
	  stage: ${env:STAGE, 'prod'}
	  timeout: 600
	  region: ${env:AWSREGION}
	  iamRoleStatements:
		  - Effect: "Allow"
			Action:
			  - "s3:*"
			Resource:
			  - "arn:aws:s3:::*/*"
	```

	```
	provider:
		name: aws
		timeout: 10
		runtime: provided
		iamRoleStatements:
			# Allow to put a file in the `my-bucket` S3 bucket
			-   Effect: Allow
				Action: s3:PutObject
				Resource: 'arn:aws:s3:::my-bucket/*'
			# Allow to query and update the `example` DynamoDB table
	```
